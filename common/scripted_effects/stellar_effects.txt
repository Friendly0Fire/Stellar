set_deployment_cost = {
	# Get the sprawl over admin cap
	if = {
		limit = {
			empire_sprawl_over_cap >= 0x100 # if >= 256, we will take it as a multiple of 16 to get up to 4080 above cap
		}
		SLEX_get_custom_parameter = {
			CONDITION_NAME = empire_sprawl_over_cap
			VARIABLE = empire_sprawl_var
			ROUNDING_ZEROES = 0
		}
	} else = {
		SLEX_get_custom_parameter = {
			CONDITION_NAME = empire_sprawl_over_cap
			VARIABLE = empire_sprawl_var

		}
	}

	if = {
		limit = {
			check_variable = {
				which = empire_sprawl_var
				value > 0
			}
		}

		set_variable = {
			which = deployment_cost_var_$ID$
			value = $ADD$
		}
		multiply_variable = {
			which = deployment_cost_var_$ID$
			value = empire_sprawl_var
		}
		change_variable = {
			which = deployment_cost_var_$ID$
			value = $BASE$
		}
	} else = {
		set_variable = {
			which = deployment_cost_var_$ID$
			value = $BASE$
		}
	}
}

pay_deployment_cost = {
	add_resource = {
		$RESOURCE$ = -1
		mult = deployment_cost_var_$ID$
	}
}
